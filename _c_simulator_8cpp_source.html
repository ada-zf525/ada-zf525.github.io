113[[nodiscard]] double ResidenceTimeSeconds(const mass_balance::SStream& feed,
114 double min_solid_volumetric_flow_m3_per_s)
115{
116 const double total_solid_mass_flow_kg_per_s = feed.TotalMassKgPerSecond();
117
118 const double solid_volumetric_flow_m3_per_s = std::max(
119 total_solid_mass_flow_kg_per_s / kSolidDensityKgPerM3, min_solid_volumetric_flow_m3_per_s);
120
121 return (kSolidsVolumeFraction * kUnitVolumeM3) / solid_volumetric_flow_m3_per_s;
122}

135[[nodiscard]] double Recovery(double kinetic_constant_per_s, double residence_time_s)
136{
137 const double kinetic_residence_product = kinetic_constant_per_s * residence_time_s;
138
139 return kinetic_residence_product / (1.0 + kinetic_residence_product);
140}

165[[nodiscard]] std::vector<mass_balance::SStream>
166SplitUnitFeed(const mass_balance::SStream& feed, int output_count,
167 const mass_balance::SSimulationOptions& options)
168{
169 const double residence_time_s =
170 ResidenceTimeSeconds(feed, options.min_solid_volumetric_flow_m3_per_s);
171
172 if (output_count == kTypeAOutputCount) {
173 mass_balance::SStream concentrate;
174 mass_balance::SStream tailings;
175
176 for (int component = 0; component < mass_balance::kComponentCount; ++component) {
177 const double recovery = Recovery(kTypeAKinetic[component], residence_time_s);
178
179 concentrate.component_kg_per_s[component] =
180 feed.component_kg_per_s[component] * recovery;
181
182 tailings.component_kg_per_s[component] =
183 feed.component_kg_per_s[component] * (1.0 - recovery);
184 }
185
186 return {concentrate, tailings};
187 }
188
189 if (output_count == kTypeBOutputCount) {
190 mass_balance::SStream concentrate_1;
191 mass_balance::SStream concentrate_2;
192 mass_balance::SStream tailings;
193
194 for (int component = 0; component < mass_balance::kComponentCount; ++component) {
195 const double k1_tau = kTypeBConcentrate1Kinetic[component] * residence_time_s;
196 const double k2_tau = kTypeBConcentrate2Kinetic[component] * residence_time_s;
197
198 /*
199 * Type B units have two valuable concentrate outputs competing for
200 * the same component. The latest project brief defines the recovery
201 * to each concentrate stream using a shared denominator:
202 *
203 * R_i^Cn = k_i^n tau / (1 + sum_n k_i^n tau)
204 *
205 * For two Type B concentrate streams:
206 *
207 * R1 = k1 tau / (1 + k1 tau + k2 tau)
208 * R2 = k2 tau / (1 + k1 tau + k2 tau)
209 *
210 * This guarantees R1 + R2 < 1 without an artificial post-hoc clamp.
211 */
212 const double denominator = 1.0 + k1_tau + k2_tau;
213
214 const double recovery_1 = k1_tau / denominator;
215 const double recovery_2 = k2_tau / denominator;
216
217 concentrate_1.component_kg_per_s[component] =
218 feed.component_kg_per_s[component] * recovery_1;
219
220 concentrate_2.component_kg_per_s[component] =
221 feed.component_kg_per_s[component] * recovery_2;
222
223 tailings.component_kg_per_s[component] =
224 feed.component_kg_per_s[component] * (1.0 - recovery_1 - recovery_2);
225 }
226
227 return {concentrate_1, concentrate_2, tailings};
228 }
229
230 return {};
231}

240[[nodiscard]] double MaxRelativeDifference(const mass_balance::SStream& current,
241 const mass_balance::SStream& previous)
242{
243 constexpr double kSmallDenominator = 1.0e-12;
244
245 double max_relative_difference = 0.0;
246
247 for (int component = 0; component < mass_balance::kComponentCount; ++component) {
248 const double numerator = std::abs(current.component_kg_per_s[component] -
249 previous.component_kg_per_s[component]);
250
251 const double denominator =
252 std::max(std::abs(previous.component_kg_per_s[component]), kSmallDenominator);
253
254 max_relative_difference = std::max(max_relative_difference, numerator / denominator);
255 }
256
257 return max_relative_difference;
258}

270[[nodiscard]] bool RouteStream(std::span<const int> circuit_span, int destination,
271 const mass_balance::SStream& stream,
272 std::vector<mass_balance::SStream>& next_unit_feeds,
273 mass_balance::SProductStreams& products)
274{
275 const int n_units = cv_num_units(circuit_span);
276
277 if (cv_is_unit_dest(circuit_span, destination)) {
278 next_unit_feeds[static_cast<std::size_t>(destination)].Add(stream);
279 return true;
280 }
281
282 if (cv_is_tailings_dest(circuit_span, destination)) {
283 products.tailings.Add(stream);
284 return true;
285 }
286
287 if (cv_is_product_dest(circuit_span, destination)) {
288 const int product_index = destination - n_units;
289
290 if (product_index == 0) {
291 products.palusznium_product.Add(stream);
292 return true;
293 }
294
295 if (product_index == 1) {
296 products.gormanium_product.Add(stream);
297 return true;
298 }
299 }
300
301 return false;
302}

343mass_balance::SSimulationResult CSimulator::Simulate(std::span<const int> circuit_span) const
344{
345 mass_balance::SSimulationResult result;
346
347 if (!cv_has_valid_shape(circuit_span)) {
348 result.failure_reason = "Circuit vector failed GATypedefs structural validation.";
349 return result;
350 }
351
352 result.valid_shape = true;
353
354 const int n_units = cv_num_units(circuit_span);
355 const mass_balance::SStream external_feed = options_.feed;
356
357 std::vector<mass_balance::SStream> unit_feeds(static_cast<std::size_t>(n_units));
358
359 /*
360 * Initial guess:
361 *
362 * The external feed enters the declared feed unit. Recycle feeds are
363 * initially zero and are built up through fixed-point iteration.
364 */
365 unit_feeds[static_cast<std::size_t>(cv_feed_dest(circuit_span))].Add(external_feed);
366
367 mass_balance::SProductStreams products;
368
369 for (int iteration = 1; iteration <= options_.max_iterations; ++iteration) {
370 const std::vector<mass_balance::SStream> old_unit_feeds = unit_feeds;
371
372 std::vector<std::vector<mass_balance::SStream>> unit_outputs(
373 static_cast<std::size_t>(n_units));
374
375 for (int unit = 0; unit < n_units; ++unit) {
376 unit_outputs[static_cast<std::size_t>(unit)] =
377 SplitUnitFeed(old_unit_feeds[static_cast<std::size_t>(unit)],
378 cv_unit_type(circuit_span, unit), options_);
379
380 if (unit_outputs[static_cast<std::size_t>(unit)].empty()) {
381 result.iterations = iteration;
382 result.failure_reason = "Unrecognised unit output count.";
383 return result;
384 }
385 }
386
387 std::vector<mass_balance::SStream> next_unit_feeds(static_cast<std::size_t>(n_units));
388 mass_balance::SProductStreams next_products;
389
390 /*
391 * Re-inject the external feed every iteration. This keeps the iteration
392 * equation as:
393 *
394 * next feeds = external feed contribution + routed internal streams
395 */
396 next_unit_feeds[static_cast<std::size_t>(cv_feed_dest(circuit_span))].Add(external_feed);
397
398 for (int unit = 0; unit < n_units; ++unit) {
399 const int output_count = cv_unit_type(circuit_span, unit);
400
401 for (int output = 0; output < output_count; ++output) {
402 const int destination = cv_unit_dest(circuit_span, unit, output);
403
404 const bool routed = RouteStream(
405 circuit_span, destination,
406 unit_outputs[static_cast<std::size_t>(unit)][static_cast<std::size_t>(output)],
407 next_unit_feeds, next_products);
408
409 if (!routed) {
410 result.iterations = iteration;
411 result.failure_reason = "Encoded stream destination could not be routed.";
412 return result;
413 }
414 }
415 }
416
417 double max_relative_change = 0.0;
418 for (int unit = 0; unit < n_units; ++unit) {
419 max_relative_change =
420 std::max(max_relative_change,
421 MaxRelativeDifference(next_unit_feeds[static_cast<std::size_t>(unit)],
422 old_unit_feeds[static_cast<std::size_t>(unit)]));
423 }
424
425 unit_feeds = std::move(next_unit_feeds);
426 products = next_products;
427
428 result.iterations = iteration;
429 result.max_relative_change = max_relative_change;
430 result.products = products;
431 result.mass_residual_kg_per_s = MassResidualKgPerSecond(external_feed, products);
432
433 if (!ProductsAreFinite(products) || !std::isfinite(result.mass_residual_kg_per_s)) {
434 result.failure_reason = "Non-finite stream value encountered during simulation.";
435 return result;
436 }
437
438 if (max_relative_change < options_.relative_tolerance) {
439 result.converged = true;
440
441 const CEconomicModel economic_model{economic_parameters_};
442 result.value_per_s = economic_model.Evaluate(circuit_span, products);
443
444 return result;
445 }
446 }
447
448 result.failure_reason = "Mass-balance fixed-point iteration did not converge.";
449 return result;
450}

452double CSimulator::EvaluatePerformance(std::span<const int> circuit_span) const
