117ParseResult parse_circuit_vector(std::span<const int> circuit_vector)
118{
119 ParseResult result;
120
121 if (circuit_vector.size() < 6) {
122 result.error = "Circuit vector is too short.";
123 return result;
124 }
125
126 const int n_inputs = circuit_vector[0];
127 const int n_units = circuit_vector[1];
128 const int n_product_streams = circuit_vector[2];
129
130 if (n_inputs != 1) {
131 result.error = "Circuit vector must contain exactly one input.";
132 return result;
133 }
134 if (n_units <= 0) {
135 result.error = "Circuit vector must contain at least one unit.";
136 return result;
137 }
138 if (n_product_streams != 3) {
139 result.error = "Circuit vector must contain exactly three final output streams.";
140 return result;
141 }
142
143 const std::size_t unit_count = static_cast<std::size_t>(n_units);
144 const std::size_t header_size = 3;
145 const std::size_t unit_counts_offset = header_size;
146 const std::size_t feed_offset = unit_counts_offset + unit_count;
147 const std::size_t outputs_offset = feed_offset + static_cast<std::size_t>(n_inputs);
148
149 if (circuit_vector.size() < outputs_offset) {
150 result.error = "Circuit vector is missing unit counts or feed destination.";
151 return result;
152 }
153
154 result.spec.n_inputs = n_inputs;
155 result.spec.n_units = n_units;
156 result.spec.n_product_streams = n_product_streams;
157 result.spec.units.reserve(unit_count);
158
159 std::size_t total_outputs = 0;
160 for (std::size_t unit_id = 0; unit_id < unit_count; ++unit_id) {
161 const int output_count = circuit_vector[unit_counts_offset + unit_id];
162 if (output_count != 2 && output_count != 3) {
163 result.error = "Unit output count must be 2 for Type A or 3 for Type B.";
164 return result;
165 }
166
167 CircuitUnitSpec unit;
168 unit.id = static_cast<int>(unit_id);
169 unit.output_count = output_count;
170 unit.destinations.reserve(static_cast<std::size_t>(output_count));
171 result.spec.units.push_back(unit);
172 total_outputs += static_cast<std::size_t>(output_count);
173 }
174
175 const std::size_t expected_size = outputs_offset + total_outputs;
176 if (circuit_vector.size() != expected_size) {
177 result.error = "Circuit vector length does not match declared unit output counts.";
178 return result;
179 }
180
181 result.spec.feed_destination = circuit_vector[feed_offset];
182 if (!is_unit_destination(result.spec.feed_destination, n_units)) {
183 result.error = "Feed destination must point to a unit.";
184 return result;
185 }
186
187 std::size_t cursor = outputs_offset;
188 for (auto& unit : result.spec.units) {
189 for (int output_index = 0; output_index < unit.output_count; ++output_index) {
190 const int destination = circuit_vector[cursor++];
191 if (destination_kind(destination, n_units) == DestinationKind::Invalid) {
192 result.error = "Unit output destination is outside the valid destination range.";
193 return result;
194 }
195 unit.destinations.push_back(destination);
196 }
197 }
198
199 result.ok = true;
200 return result;
201}

213std::vector<int> encode_circuit_vector(const CircuitSpec& spec)
214{
215 assert(std::cmp_equal(spec.n_units, spec.units.size()));
216 std::vector<int> circuit_vector;
217 circuit_vector.reserve(3 + spec.units.size() + 1 + (spec.units.size() * 3));
218
219 circuit_vector.push_back(spec.n_inputs);
220 circuit_vector.push_back(spec.n_units);
221 circuit_vector.push_back(spec.n_product_streams);
222
223 for (const auto& unit : spec.units) {
224 circuit_vector.push_back(unit.output_count);
225 }
226
227 circuit_vector.push_back(spec.feed_destination);
228
229 for (const auto& unit : spec.units) {
230 for (int destination : unit.destinations) {
231 circuit_vector.push_back(destination);
232 }
233 }
