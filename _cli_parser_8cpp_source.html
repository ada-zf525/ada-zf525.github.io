86template <typename T>
87bool TakeNumeric(std::string_view flag, int& index, int argc, const char* const* argv,
88 ParseResult& result, std::optional<T>& destination)
89{
90 const char* value = TakeValue(flag, index, argc, argv, result);
91 if (value == nullptr) {
92 return false;
93 }
94 const std::string_view text{value};
95 T parsed{};
96 const auto* text_end = text.data() + text.size();
97 const auto outcome = std::from_chars(text.data(), text_end, parsed);
98 if (outcome.ec != std::errc{} || outcome.ptr != text_end) {
99 result.error = "Invalid numeric value for ";
100 result.error.append(flag);
101 return false;
102 }
103 destination = parsed;
104 return true;
105}

120std::string ValidateOptions(const CliOptions& options)
121{
122 if (options.circuit.vector_inline.has_value() && options.circuit.vector_file.has_value()) {
123 return "--vector and --vector-file are mutually exclusive";
124 }
125 if (options.case_preset != CasePreset::None && options.circuit.fixed_prefix.has_value()) {
126 return "--case and --fixed-prefix are mutually exclusive";
127 }
128 if (options.circuit.fixed_prefix.has_value() && options.mode != Mode::Optimise) {
129 return "--fixed-prefix is only valid in --mode optimise";
130 }
131 const bool needs_vector = options.mode == Mode::Evaluate || options.mode == Mode::Validate ||
132 options.mode == Mode::Visualise;
133 const bool has_vector =
134 options.circuit.vector_inline.has_value() || options.circuit.vector_file.has_value();
135 if (needs_vector && !has_vector) {
136 return "Selected --mode requires --vector or --vector-file";
137 }
138 if (options.mode == Mode::Visualise && !options.outputs.plot_image.has_value()) {
139 return "--mode visualise requires --plot";
140 }
141 if (options.ga.population_size.has_value() && *options.ga.population_size <= 0) {
142 return "--population-size must be positive";
143 }
144 if (options.ga.max_iterations.has_value() && *options.ga.max_iterations <= 0) {
145 return "--max-iterations must be positive";
146 }
147 if (options.ga.tournament_size.has_value() && *options.ga.tournament_size <= 0) {
148 return "--tournament-size must be positive";
149 }
150 if (options.ga.elitism_count.has_value() && *options.ga.elitism_count < 0) {
151 return "--elitism-count must be non-negative";
152 }
153 if (options.ga.max_mutations.has_value() && *options.ga.max_mutations < 0) {
154 return "--max-mutations must be non-negative";
155 }
156 if (options.ga.crossover_probability.has_value() &&
157 (*options.ga.crossover_probability < 0.0 || *options.ga.crossover_probability > 1.0)) {
158 return "--crossover-probability must be in [0, 1]";
159 }
160 return {};
161}

163std::optional<std::vector<int>> ParseIntegerList(std::string_view text)
164{
165 std::string normalised{text};
166 for (char& character : normalised) {
167 if (character == ',') {
168 character = ' ';
169 }
170 }
171 std::istringstream stream(normalised);
172 std::vector<int> values;
173 std::string token;
174 while (stream >> token) {
175 try {
176 std::size_t consumed = 0;
177 const int value = std::stoi(token, &consumed);
178 if (consumed != token.size()) {
179 return std::nullopt;
180 }
181 values.push_back(value);
182 } catch (...) {
183 return std::nullopt;
184 }
185 }
186 return values;
187}
