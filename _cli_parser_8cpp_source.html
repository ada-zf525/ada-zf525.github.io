69template <typename T> std::optional<T> ParseNumericValue(std::string_view text)
70{
71 const std::string token{text};
72 T parsed{};
73 const auto* text_end = token.data() + token.size();
74 const auto outcome = std::from_chars(token.data(), text_end, parsed);
75 if (outcome.ec != std::errc{} || outcome.ptr != text_end) {
76 return std::nullopt;
77 }
78 return parsed;
79}

132template <typename T>
133bool TakeNumeric(std::string_view flag, int& index, int argc, const char* const* argv,
134 ParseResult& result, std::optional<T>& destination)
135{
136 const char* value = TakeValue(flag, index, argc, argv, result);
137 if (value == nullptr) {
138 return false;
139 }
140 const std::string_view text{value};
141 const std::optional<T> parsed = ParseNumericValue<T>(text);
142 if (!parsed.has_value()) {
143 result.error = "Invalid numeric value for ";
144 result.error.append(flag);
145 return false;
146 }
147 destination = parsed;
148 return true;
149}

254std::optional<std::vector<int>> ParseIntegerList(std::string_view text)
255{
256 std::string normalised{text};
257 for (char& character : normalised) {
258 if (character == ',') {
259 character = ' ';
260 }
261 }
262 std::istringstream stream(normalised);
263 std::vector<int> values;
264 std::string token;
265 while (stream >> token) {
266 try {
267 std::size_t consumed = 0;
268 const int value = std::stoi(token, &consumed);
269 if (consumed != token.size()) {
270 return std::nullopt;
271 }
272 values.push_back(value);
273 } catch (...) {
274 return std::nullopt;
275 }
276 }
277 return values;
278}
