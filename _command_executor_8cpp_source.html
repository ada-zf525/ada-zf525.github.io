44bool ParseIntegerList(const std::string& text, std::vector<int>& destination, std::string& error)
45{
46 std::string normalised = text;
47 for (char& character : normalised) {
48 if (character == ',') {
49 character = ' ';
50 }
51 }
52 std::istringstream stream(normalised);
53 std::string token;
54 while (stream >> token) {
55 try {
56 std::size_t consumed = 0;
57 const int value = std::stoi(token, &consumed);
58 if (consumed != token.size()) {
59 error = "Invalid integer token: " + token;
60 return false;
61 }
62 destination.push_back(value);
63 } catch (...) {
64 error = "Invalid integer token: " + token;
65 return false;
66 }
67 }
68 return true;
69}

77std::string CsvEscape(std::string_view value)
78{
79 const bool must_quote = value.find_first_of(",\"\n\r") != std::string_view::npos;
80 if (!must_quote) {
81 return std::string{value};
82 }
83 std::string escaped;
84 escaped.reserve(value.size() + 2);
85 escaped.push_back('"');
86 for (const char character : value) {
87 if (character == '"') {
88 escaped.append("\"\"");
89 } else {
90 escaped.push_back(character);
91 }
92 }
93 escaped.push_back('"');
94 return escaped;
95}

117std::vector<int> MakeFixedPresetPrefix(int num_units)
118{
119 std::vector<int> fixed_prefix;
120 if (num_units <= 0) {
121 return fixed_prefix;
122 }
123
124 fixed_prefix.reserve(cv_fixed_prefix_length(num_units));
125 fixed_prefix.push_back(1);
126 fixed_prefix.push_back(num_units);
127 fixed_prefix.push_back(3);
128
129 const int type_b_count = std::min(3, num_units);
130 const int type_a_count = num_units - type_b_count;
131 fixed_prefix.insert(fixed_prefix.end(), type_a_count, 2);
132 fixed_prefix.insert(fixed_prefix.end(), type_b_count, 3);
133
134 return fixed_prefix;
135}

160std::string ValidateFixedPrefix(std::span<const int> fixed_prefix)
161{
162 if (fixed_prefix.size() < 4) {
163 return "fixed prefix must include feed count, unit count, product count, and unit types";
164 }
165 if (cv_num_feed(fixed_prefix) != 1) {
166 return "fixed prefix must contain exactly one feed";
167 }
168 if (cv_num_units(fixed_prefix) <= 0) {
169 return "fixed prefix must contain at least one unit";
170 }
171 if (cv_num_products(fixed_prefix) != 3) {
172 return "fixed prefix must contain exactly three product streams";
173 }
174
175 const std::size_t expected_size = cv_fixed_prefix_length(cv_num_units(fixed_prefix));
176 if (fixed_prefix.size() != expected_size) {
177 return "fixed prefix length does not match the declared unit count";
178 }
179
180 for (int unit_index = 0; unit_index < cv_num_units(fixed_prefix); ++unit_index) {
181 const int unit_type = cv_unit_type(fixed_prefix, unit_index);
182 if (unit_type != 2 && unit_type != 3) {
183 return "fixed prefix unit types must be 2 or 3";
184 }
185 }
186
187 return {};
188}

340LoadedCircuit LoadCircuitVector(const CircuitInputs& inputs)
341{
342 LoadedCircuit outcome;
343 if (inputs.vector_inline.has_value()) {
344 outcome.values = *inputs.vector_inline;
345 outcome.ok = true;
346 return outcome;
347 }
348 if (inputs.vector_file.has_value()) {
349 std::ifstream file(*inputs.vector_file);

351 outcome.error = "Cannot open vector file: " + *inputs.vector_file;
352 return outcome;

354 std::ostringstream buffer;
355 buffer << file.rdbuf();
356 outcome.ok = ParseIntegerList(buffer.str(), outcome.values, outcome.error);
357 return outcome;

375int OptimiseExecutor::Execute(const CliOptions& options)
376{
377 if (options.ga.rng_seed.has_value()) {
378 seed_rngs(*options.ga.rng_seed);
379 }
380
381 const Algorithm_Parameters ga_parameters = BuildAlgorithmParameters(options.ga);
382 const SA_Parameters sa_parameters = BuildSaParameters(options.sa);
383 const bool use_sa = options.method == OptimisationMethod::SimulatedAnnealing;
384 const char* method_label = use_sa ? "sa" : "ga";
385
386 GA_Functions ga_functions{};
387 ga_functions.vector_extent = &cv_variable_length;
388 ga_functions.vector_crossover = &unit_block_crossover;
389
390 const CSimulator simulator{BuildSimulationOptions(options.economic),
391 BuildEconomicParameters(options.economic)};
392
393 const FitnessFunction fitness = [&simulator](std::span<const int> circuit) {
394 return simulator.EvaluatePerformance(circuit);
395 };
396
397 const ValidityFunction validity = [](std::span<const int> circuit) {
398 return check_validity(circuit);
399 };
400
401 const auto start = std::chrono::steady_clock::now();
402
403 std::vector<int> best_circuit_values;
404 double best_performance = -std::numeric_limits<double>::infinity();
405
406 if (options.case_preset == CasePreset::Swappable && !options.circuit.fixed_prefix.has_value()) {
407 const int swappable_units = ResolveUnitCount(options, 8);
408 if (use_sa) {
409 BaseResult<std::span<int>> result = optimize_swappable_span_sa(
410 swappable_units, fitness, validity, ga_functions, sa_parameters);
411 best_circuit_values.assign(result.data.begin(), result.data.end());
412 best_performance = result.performance;
413 } else {
414 BaseResult<std::span<int>> result = optimize_swappable_span(
415 swappable_units, fitness, validity, ga_functions, ga_parameters);
416 best_circuit_values.assign(result.data.begin(), result.data.end());
417 best_performance = result.performance;
418 }
419 } else {
420 std::vector<int> fixed_prefix = ResolveFixedPrefix(options);
421 if (fixed_prefix.empty()) {
422 log_stream_ << "Optimise failed: no fixed prefix (use --case or --fixed-prefix)\n";
423 return 2;
424 }
425 const std::string fixed_prefix_error = ValidateFixedPrefix(fixed_prefix);
426 if (!fixed_prefix_error.empty()) {
427 log_stream_ << "Optimise failed: " << fixed_prefix_error << '\n';
428 return 2;
429 }
430
431 if (use_sa) {
432 BaseResult<std::span<int>> result = optimize_span_sa(
433 std::span<const int>(fixed_prefix), fitness, validity, ga_functions, sa_parameters);
434 best_circuit_values.assign(result.data.begin(), result.data.end());
435 best_performance = result.performance;
436 } else {
437 BaseResult<std::span<int>> result = optimize_span(
438 std::span<const int>(fixed_prefix), fitness, validity, ga_functions, ga_parameters);
439 best_circuit_values.assign(result.data.begin(), result.data.end());
440 best_performance = result.performance;
441 }
442 }
443
444 const auto end = std::chrono::steady_clock::now();
445 const double runtime_seconds = std::chrono::duration<double>(end - start).count();
446
447 const std::span<const int> best_circuit{best_circuit_values};
448 const bool valid = std::isfinite(best_performance);
449 if (options.outputs.plot_image.has_value()) {
450 plot_span(best_circuit, options.outputs.plot_image->c_str());
451 }
452
453 result_stream_ << "mode,method,circuit,performance,valid,runtime_seconds\n";
454 result_stream_ << "optimise," << method_label << ','
455 << CsvEscape(FormatIntegerVector(best_circuit)) << ',' << best_performance << ','
456 << (valid ? "true" : "false") << ',' << runtime_seconds << '\n';
457
458 log_stream_ << "method = " << method_label << ", best performance = " << best_performance
459 << " (runtime " << runtime_seconds << "s)\n";
460 return valid ? 0 : 3;
461}
